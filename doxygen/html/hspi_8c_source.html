135 void hspi_writeFIFO(uint8_t *write_data, uint16_t bytes)
136 {
137  uint8_t word_ind = 0;
138 
139  if(bytes > HSPI_FIFO_SIZE) // TODO set error status
140  return;
141 
142  hspi_setBits(bytes); // Update FIFO with number of bits we will send
143 
144 // First do a fast write with 32 bit chunks at a time
145  while(bytes >= 4)
146  {
147 // Cast both source and destination to 4 byte word pointers
148  ((uint32_t *)SPI_FLASH_C0(HSPI)) [word_ind] = \
149  ((uint32_t *)write_data) [word_ind];
150 
151  bytes -= 4;
152  word_ind++;
153  }
154 
155 // Next write remaining bytes (if any) to FIFO as a single word
156 // Note: We follow the good practice of avoiding access past the end of an array.
157 // (ie. protected memory, PIC using retw from flash ... could crash some CPU's)
158  if(bytes) // Valid counts are now: 3,2,1 bytes
159  {
160  uint32_t last_word = 0; // Last word to send
161  uint16_t byte_ind = word_ind << 2; // Convert to Byte index
162 
163 // Working MSB to LSB allows assigment to LSB without shifting
164 // Compiler can optimize powers of 8 into byte swaps, etc... (on some CPU's)
165  while(bytes--) // index is now 2,1,0 matching required storage offset
166  {
167  last_word <<= 8;
168 // 2,1,0
169  last_word |= ((uint32_t) 0xffU & write_data[byte_ind + bytes ]);
170  }
171 // Send last partial word
172  ((uint32_t *)SPI_FLASH_C0(HSPI)) [word_ind] = last_word;
173  }
174 }

309 void hspi_TxBuffered(uint8_t *write_data, uint32_t bytes, uint32_t repeats)
310 {
311  uint16_t d_ind = 0; // data index, size of write_data (bytes)
312  uint16_t f_ind = 0; // fifo buffer index
313  uint8_t buffer[HSPI_FIFO_SIZE]; // fifo buffer, same as HSPI FIFO
314 
315  if ((bytes < 1) || (repeats < 1))
316  return; // Error parameter
317 
318  while(repeats)
319  {
320  buffer[f_ind++] = write_data[d_ind++];
321  if(d_ind >= bytes) // size of data
322  {
323  d_ind = 0;
324  --repeats; // data has been copied once
325  }
326 // Send a full FIFO block - or - remaining data
327  if(!repeats || f_ind >= HSPI_FIFO_SIZE)
328  {
329  hspi_Tx(buffer,f_ind);
330  f_ind = 0;
331  }
332  }
