473 ICACHE_FLASH_ATTR
474 void uart_config(uint8 uart_no, uint32_t baud, uint8_t data_bits, uint8_t stop_bits, uint8_t parity)
475 {
476  if (uart_no == UART0)
477  {
478  PIN_PULLUP_DIS(PERIPHS_IO_MUX_U0TXD_U);
479  PIN_FUNC_SELECT(PERIPHS_IO_MUX_U0TXD_U, FUNC_U0TXD);
480  PIN_PULLUP_DIS (PERIPHS_IO_MUX_U0RXD_U);
481  PIN_FUNC_SELECT(PERIPHS_IO_MUX_U0RXD_U, FUNC_U0RXD);
482  }
483  else
484  {
485 // GPIO2
486  PIN_PULLUP_DIS(PERIPHS_IO_MUX_GPIO2_U);
487  PIN_FUNC_SELECT(PERIPHS_IO_MUX_GPIO2_U, FUNC_U1TXD_BK);
488  }
489 
490  uart_div_modify(uart_no, UART_CLK_FREQ / baud);
491 
492 // Data bits 5..8
493  data_bits = data_bits - 5;
494 
495 // STOP BITS 1..2
496  if(stop_bits == 2)
497  stop_bits = TWO_STOP_BIT;
498  else
499  stop_bits = ONE_STOP_BIT;
500 
501 // PARITY NO_PARITY,ODD_PARITY,EVEN_PARITY
502 
503  WRITE_PERI_REG(UART_CONF0(uart_no),
504  ((data_bits & UART_BIT_NUM) << UART_BIT_NUM_S) |
505  ((stop_bits & UART_STOP_BIT_NUM) << UART_STOP_BIT_NUM_S) |
506  parity);
507 
508 //clear rx and tx fifo,not ready
509 //RESET FIFO
510  tx_fifo_flush(uart_no);
511  rx_fifo_flush(uart_no);
512 
513  if (uart_no == UART0)
514  {
515  WRITE_PERI_REG(UART_CONF1(uart_no),
516  (( 2 & UART_RX_TOUT_THRHD) << UART_RX_TOUT_THRHD_S) | UART_RX_TOUT_EN |
517  ((16 & UART_RXFIFO_FULL_THRHD) << UART_RXFIFO_FULL_THRHD_S) |
518  ((64 & UART_TXFIFO_EMPTY_THRHD) << UART_TXFIFO_EMPTY_THRHD_S) );
519  WRITE_PERI_REG(UART_INT_CLR(uart_no), 0xffff);
520  WRITE_PERI_REG(UART_INT_ENA(uart_no), UART_RXFIFO_TOUT_INT_ENA | UART_RXFIFO_FULL_INT_ENA);
521  }
522 }
