74 uint32_t tft_abs_window(int16_t x, int16_t y, int16_t w, int16_t h)
75 {
76  uint8_t tmp[4];
77 
78  uint16_t xl,yl;
79 
80  int16_t ww,hh;
81  uint32_t bytes;
82 
83  // Check for basic out of bounds conditions
84 
85  // Width or Height <= 0 ? Bogus range
86  if(w <= 0 || h <= 0)
87  return(0);
88 
89  // X >= Width ??
90  if(x >= tft->w)
91  return(0);
92 
93  // Y >= Height ??
94  if(y >= tft->h)
95  return(0);
96 
97  // Clipping tests for < 0
98  // X < 0 ? Clip
99  if(x < 0)
100  x = 0;
101  // Y < 0 ? Clip
102  if(y < 0)
103  y = 0;
104 
105  // Clip check X + W
106  ww = (x + w) - tft->w;
107  if(ww > 0)
108  w -= ww;
109  if(w <= 0)
110  return(0);
111 
112  // Clip check Y + H
113  hh = (y + h) - tft->h;
114  if(hh > 0)
115  h -= hh;
116  if(h <= 0)
117  return(0);
118 
119  // Now We know the result will fit
120  xl = x + w - 1;
121  yl = y + h - 1;
122 
123  tmp[0] = x >> 8;
124  tmp[1] = x & 0xff;
125  tmp[2] = xl >> 8;
126  tmp[3] = xl & 0xff;
127  tft_writeCmdData(0x2A, tmp, 4);
128  tmp[0] = y >> 8;
129  tmp[1] = y & 0xff;
130  tmp[2] = yl >> 8;
131  tmp[3] = yl & 0xff;
132  tft_writeCmdData(0x2B, tmp, 4);
133 
134  bytes = w;
135  bytes *= h;
136  return( bytes);
137 }
