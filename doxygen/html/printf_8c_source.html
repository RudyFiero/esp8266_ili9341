560 static int etoa(double x,char *str, int prec, int sign)
561 {
562  double scale; /* scale factor */
563  int i, /* counter */
564  d, /* a digit */
565  expon; /* exponent */
566  char *base = str;
567 
568  scale = 1.0 ; /* scale = 10 ** prec */
569  i = prec ;
570  while ( i-- )
571  scale *= 10.0 ;
572  if ( x == 0.0 )
573  {
574  expon = 0 ;
575  scale *= 10.0 ;
576  }
577  else
578  {
579  expon = prec ;
580  if ( x < 0.0 )
581  {
582  *str++ = '-' ;
583  x = -x ;
584  }
585  else
586  {
587  if(sign == 2)
588  *str++ = '+' ;
589  else
590  *str++ = ' ' ;
591  }
592  if ( x > scale )
593  {
594 
595  /* need: scale<x<scale*10 */
596  scale *= 10.0 ;
597  while ( x >= scale )
598  {
599  x /= 10.0 ;
600  ++expon ;
601  }
602  }
603  else
604  {
605  while ( x < scale )
606  {
607  x *= 10.0 ;
608  --expon ;
609  }
610  scale *= 10.0 ;
611  }
612  /* at this point, .1*scale <= x < scale */
613  x += 0.5 ; /* round */
614  if ( x >= scale )
615  {
616  x /= 10.0 ;
617  ++expon ;
618  }
619  }
620  i = 0 ;
621  while ( i <= prec )
622  {
623  scale = floor( 0.5 + scale * 0.1 ) ;
624  /* now, scale <= x < 10*scale */
625  d = (int) ( x / scale ) ;
626  *str++ = d + '0' ;
627  x -= (double)d * scale ;
628  if ( i++ ) continue ;
629  *str++ = '.' ;
630  }
631  *str++ = 'e' ;
632  if ( expon < 0 )
633  {
634  expon = -expon;
635  *str++ = '-';
636  }
637  else
638  {
639  *str++ = '+';
640  }
641  *str++ = '0' + expon/10 ;
642  *str++ = '0' + expon % 10 ;
643  *str = 0;
644  return(str - base);
645 }
